1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.host }}
  {{- range $.Values.ingress.tls }}
  {{- range .hosts }}
  {{- if eq $host . }}
  https://{{ $host }}{{ $.Values.ingress.path }}
  {{- end }}
  {{- end }}
  {{- else }}
  http://{{ $host }}{{ $.Values.ingress.path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "content-chart.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch its status by running:
        kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "content-chart.fullname" . }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "content-chart.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "content-chart.fullname" . }} 8080:{{ .Values.service.port }}
  echo "Visit http://127.0.0.1:8080 to access the content service"
{{- end }}

2. Check deployment status:
  kubectl get deployment --namespace {{ .Release.Namespace }} {{ include "content-chart.fullname" . }}

3. View logs:
  kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "content-chart.name" . }} -f

{{- if .Values.persistence.enabled }}
4. Persistent storage is mounted at {{ .Values.persistence.mountPath }} using PVC {{ include "content-chart.pvcName" . }}.
{{- end }}
